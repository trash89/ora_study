-- moving lobs and vararrays

alter table OE.LINEITEM_TABLE move lob (SYS_XDBPD$) store as ( tablespace EXAMPLE);

ALTER TABLE OE.ACTION_TABLE MOVE TABLESPACE users 
    VARRAY SYS_XDBPD$ STORE AS LOB(TABLESPACE EXAMPLE);

ALTER TABLE OE.LINEITEM_TABLE MOVE TABLESPACE users 
    VARRAY "PART"."SYS_XDBPD$"  STORE AS LOB(TABLESPACE EXAMPLE)
    VARRAY "SYS_XDBPD$"         STORE AS LOB(TABLESPACE EXAMPLE);

ALTER TABLE OE.PURCHASEORDER MOVE TABLESPACE EXAMPLE 
    VARRAY "XMLDATA"."SYS_XDBPD$" STORE AS LOB(TABLESPACE EXAMPLE);

ALTER TABLE OE.PURCHASEORDER MOVE TABLESPACE users 
    VARRAY "XMLDATA"."LINEITEMS"."SYS_XDBPD$"               STORE AS LOB(TABLESPACE EXAMPLE)
    VARRAY "XMLDATA"."SHIPPING_INSTRUCTIONS"."SYS_XDBPD$"   STORE AS LOB(TABLESPACE EXAMPLE)
    VARRAY "XMLDATA"."REJECTION"."SYS_XDBPD$"               STORE AS LOB(TABLESPACE EXAMPLE)
    VARRAY "XMLDATA"."ACTIONS"."SYS_XDBPD$"                 STORE AS LOB(TABLESPACE EXAMPLE)
    VARRAY "XMLEXTRA"."EXTRADATA"                           STORE AS LOB(TABLESPACE EXAMPLE)
    VARRAY "XMLEXTRA"."NAMESPACES"                          STORE AS LOB(TABLESPACE EXAMPLE)
;
